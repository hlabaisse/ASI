import os
import struct
import numpy as np
import warnings
warnings.filterwarnings('ignore')

#Code inspired from the Internet

def read(dataset = "training", path = "."):

    if dataset is "training":
        fname_img = os.path.join(path, 'train-images.idx3-ubyte')
        fname_lbl = os.path.join(path, 'train-labels.idx1-ubyte')
    elif dataset is "testing":
        fname_img = os.path.join(path, 't10k-images.idx3-ubyte')
        fname_lbl = os.path.join(path, 't10k-labels.idx1-ubyte')

    # Load everything in some numpy arrays
    with open(fname_lbl, 'rb') as flbl:
        magic, num = struct.unpack(">II", flbl.read(8))
        lbl = np.fromfile(flbl, dtype=np.int8)

    with open(fname_img, 'rb') as fimg:
        magic, num, rows, cols = struct.unpack(">IIII", fimg.read(16))
        img = np.fromfile(fimg, dtype=np.uint8).reshape(len(lbl), rows, cols)
        
    return lbl, img

import operator
import functools
import math

#Compute the gaussian distribution
def gaussian(x, mean, std):
    #I have to take into account this outlier because I would have a math error in the exponential otherwise.
    if std == 0:  
        #I put one because the probability that x is equal to mean is very high.
        return 1
    else:
        return np.exp((-np.power(x-mean, 2.)/(2*np.power(std, 2.))))*(1 / (math.sqrt(2*math.pi) * std))

#Return the logarithm of the numerator of naive bayes
def naive_bayes(x, means, stds, k):
    gaussian_list = [gaussian(x[i], means[i], stds[i]) for i in range(len(x))]
    for i in range(len(gaussian_list)):
        #I have to deal with this particular case because I cannot apply math.log to a negative value.
        if gaussian_list[i] <= 0:
            #I put 0 because it is not regular and I have better result with 0.
            gaussian_list[i] = 0
        else:
            #I put the log because otherwise my product is equal to 0.
            gaussian_list[i] = math.log(gaussian_list[i])
            #except for binary data, I don't do it
            #gaussian_list[i] = gaussian_list[i]
    #Add (not multiply because I took the log) every component of the matrix and also the prior
    pred = (abs(functools.reduce(lambda x, y: x+y, gaussian_list)) + math.log(prior[k]))
    #only for binary data because I don't use the log:
    #pred = abs(functools.reduce(lambda x, y: x*y, gaussian_list) * prior[k])
    return pred

#Return a vector of probability to belong to each label
def loglikelihood(x, means, stds):
    expPred = []
    for k in range(10):
        #I divide by 100 because the input in my exponential is too big and is often close to 1000
        pred = naive_bayes(x, means[k], stds[k],k)/100
        #only for binary data:
        #expPred.append(pred) 
        expPred.append(np.exp(-pred))
    expSum = sum(expPred)
    for i in range(10):
        expPred[i] = expPred[i] /expSum
    return expPred

#Take the maximum label to make a prediction about the true label with his probability
def guess_label(x, means, stds, label):
    expPred =loglikelihood(x, means, stds)
    proba = max(expPred)
    g_label = expPred.index(max(expPred)) 
    return g_label, proba, label

#Compute the accuracy of my prediction
def accuracy(L, lbl):
    S = 0
    for i in range(len(L)):
        if L[i] == lbl[i]:
            S += 1
    return S/len(L)*100

3.b Describe a positive and a negative feature of the classifier for these tasks¶

A negative aspect of this classifier is that it makes a big assumption about the repartition of our data. It supposes that the shape of our data distribution is a gaussian but it is not necessary the case and so the result could be bad if the assumption is violated.
A positive aspect is that it is simple to build, we only need to implement a normal distribution. Moreover it doesn't require a big dataset to be trained.

3.c Describe any data pre-processing that you suggest for this data and your classifier¶

A possibility to improve our algorithm would be to pre-process our data as binary value for each pixel. We set a threshold, for example 128: if the pixel value is over we said it is equal to 1 and if it is under, equal to 0.
Another thing could be to observe the shape repartition of our data. Indeed we assume it was normal distribution but in most of the case it is not true. Thus after observing some shape repartition we can model it by a function that fits it better.
An assumption we make with Naive Bayesian Classifier is that features are independent but it is not the case. Thus, we should implement dependency in our data pre-processing.
Another thing we can do is to gather some pixels by blocks. Indeed when there is ink on a pixel of paper, there is probably ink on the next pixel. Grouping them would save time computation and skip redundant features.

#Implement the function to estimate the weights of the linear regression
def lsfit(img, lbl):
    a = np.dot(np.transpose(img), img)
    b = np.dot(np.transpose(img), lbl)
    #I use pinv because using inv because a is only pseudo-inversible
    c = pinv(a)
    return np.dot(c, b)

import math as m
#Compute the mean_square error for 
def mean_squared_error(y_test, label):
    E = [0]*10
    E_tot = 0
    N = [0]*10
    for i in range(len(y_test)):
        for j in range(10):
            if label[i] == j:
                E_tot += (y_test[i] - label[i])**2
                E[j] += (y_test[i] - label[i])**2
                N[j] += 1
    #I take the square root to plot it after
    E = [m.sqrt(E[k] / N[k]) for k in range(10)]
    E_tot = E_tot / len(y_test)
    return E, E_tot

    if y > 9:
        #in case the declared label is over 9, I put 9 because we don't have digits bigger than 9
        label = 9
        return label
    else:
        #label is the closer integer to y
        label = int(round(y))
        return label

4.e Discuss regression performance with respect to classification performance¶

Both on the MNIST data and the CIPHAR10 the results are better for the bayesian classifier. Indeed, the accuracy is around 25% and 11% for the linear regarding the 65% and 27% for the classifier. It is only a bit better than a random classifier for the CIPHAR10. With the confusion matrix we see that the labels are balanced, it is difficult to guess all labels.

4.f Describe one limitation of using regression for this particular task.¶

One limitation is that it takes into account pixel individually but not together. However we know (in particular for CIFAR) that to recognize a car or a digit it is important to understand the shape of the image and thus how pixels are related.
Moreover another limitation is the number of pixels regarding the labels. Indeed, the weight matrix has a size of the number of pixel (around 750 for the MNIST data and 3000 for the CIPHAR10) and the gap between each label is very small (1) so it could make mistakes due to this difference.
