def lsfit(year,time):
    xmean = np.mean(year)
    tmean = np.mean(time)
    xtb = np.dot(year,time)/len(year)
    x2b = np.dot(year,year)/len(year)
    w1 = (xtb - xmean*tmean)/(x2b-xmean**2)
    w0 = tmean - w1*xmean
    newVect = np.add(np.add(time,-w0),np.dot(-w1,year))
    loss = np.dot(newVect,newVect)/len(newVect)
    
    return w0,w1, loss

    err=0
    for i in range(len(x)):
        if i==len(x)-1 :
            x_reduced = x[:i]
            tt_reduced = tt[:i]
        else:

            tt_reduced = np.append(tt[:i],tt[i+1:])
        pred = makePrediction(x_reduced,x[i],tt_reduced,k)[0]
        err+=(pred-tt[i])**2
